91 {

93 
94 start:
95 
96  switch (self->_state)
97  {
98  case BC_HDC2080_STATE_ERROR:
99  {
100  self->_humidity_valid = false;
101 
102  if (self->_event_handler != NULL)
103  {
104  self->_event_handler(self, BC_HDC2080_EVENT_ERROR, self->_event_param);
105  }
106 
107  self->_state = BC_HDC2080_STATE_INITIALIZE;
108 
109  bc_scheduler_plan_current_relative(self->_update_interval);
110 
111  return;
112  }
113  case BC_HDC2080_STATE_INITIALIZE:
114  {
115  self->_state = BC_HDC2080_STATE_ERROR;
116 
117  if (!bc_i2c_write_8b(self->_i2c_channel, self->_i2c_address, 0x0e, 0x80))
118  {
119  goto start;
120  }
121 
122  self->_state = BC_HDC2080_STATE_MEASURE;
123 
124  bc_scheduler_plan_current_relative(BC_HDC2080_DELAY_INITIALIZATION);
125 
126  return;
127  }
128  case BC_HDC2080_STATE_MEASURE:
129  {
130  self->_state = BC_HDC2080_STATE_ERROR;
131 
132  if (!bc_i2c_write_8b(self->_i2c_channel, self->_i2c_address, 0x0f, 0x07))
133  {
134  goto start;
135  }
136 
137  self->_state = BC_HDC2080_STATE_READ;
138 
139  bc_scheduler_plan_current_relative(BC_HDC2080_DELAY_MEASUREMENT);
140 
141  return;
142  }
143  case BC_HDC2080_STATE_READ:
144  {
145  self->_state = BC_HDC2080_STATE_ERROR;
146 
147  uint8_t reg_interrupt;
148 
149  if (!bc_i2c_read_8b(self->_i2c_channel, self->_i2c_address, 0x04, &reg_interrupt))
150  {
151  goto start;
152  }
153 
154  if ((reg_interrupt & 0x80) == 0)
155  {
156  goto start;
157  }
158 
159  if (!bc_i2c_read_16b(self->_i2c_channel, self->_i2c_address, 0x02, &self->_reg_humidity))
160  {
161  goto start;
162  }
163 
164  if (!bc_i2c_read_16b(self->_i2c_channel, self->_i2c_address, 0x00, &self->_reg_temperature))
165  {
166  goto start;
167  }
168 
169  self->_reg_temperature = self->_reg_temperature << 8 | self->_reg_temperature >> 8;
170 
171  self->_reg_humidity = self->_reg_humidity << 8 | self->_reg_humidity >> 8;
172 
173  self->_temperature_valid = true;
174 
175  self->_humidity_valid = true;
176 
177  self->_state = BC_HDC2080_STATE_UPDATE;
178 
179  goto start;
180  }
181  case BC_HDC2080_STATE_UPDATE:
