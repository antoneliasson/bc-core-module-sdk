189 void bc_module_lcd_update(void)
190 {
191 
192  bc_module_lcd_t *self = &_bc_module_lcd;
193 
194  uint8_t spi_data[2];
195  uint8_t line;
196  uint32_t offs;
197 
198  // Send MODE
199  spi_data[0] = 0xd0;
200  spi_data[1] = 0x00;
201  bc_spi_transfer(spi_data, NULL, 2);
202 
203  self->framebuffer[0] = 0x80;
204 
205  // Address lines
206  for (line = 0x01, offs = 1; line <= 128; line++, offs += 18) {
207  // Fill the gate line addresses on the exact place in the buffer
208  self->framebuffer[offs] = reverse2(line);
209  }
210 
211  bc_spi_transfer(&self->framebuffer[0], NULL, BC_LCD_FRAMEBUFFER_SIZE);
212 }
