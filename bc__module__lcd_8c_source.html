42 void bc_module_lcd_init(bc_module_lcd_framebuffer_t *framebuffer)
43 {
44  bc_module_lcd_t *self = &_bc_module_lcd;
45 
46  bc_tca9534a_init(&_bc_module_lcd.tca9534a, BC_I2C_I2C0, 0x39);
47  bc_tca9534a_set_port_direction(&_bc_module_lcd.tca9534a, 0x00);
48  _bc_module_lcd.gpio = _BC_MODULE_LCD_DISP_CS | _BC_MODULE_LCD_DISP_ON | _BC_MODULE_LCD_LED_GREEN | _BC_MODULE_LCD_LED_RED | _BC_MODULE_LCD_LED_BLUE;
49  bc_tca9534a_write_port(&_bc_module_lcd.tca9534a, _bc_module_lcd.gpio);
50 
51  bc_spi_init(BC_SPI_SPEED_2_MHZ, BC_SPI_MODE_0);
52 
53  self->framebuffer = framebuffer->framebuffer;
54 
55  _bc_module_lcd.font = &Font;
56 
57  // Address lines
58  uint8_t line;
59  uint32_t offs;
60  for (line = 0x01, offs = 1; line <= 128; line++, offs += 18) {
61  // Fill the gate line addresses on the exact place in the buffer
62  self->framebuffer[offs] = reverse2(line);
63  }
64 
65  // Prepare buffer so the background is "white" reflective
66  bc_module_lcd_clear();
67

97 void bc_module_lcd_draw_pixel(int x, int y, bool value)
98 {
99  if(x > 127 || y > 127 || x < 0 || y < 0)
100  {
101  return;
102  }
103 
104  bc_module_lcd_t *self = &_bc_module_lcd;
105 
106  int tmp;
107 
108  switch(self->rotation)
109  {
110  case BC_MODULE_LCD_ROTATION_90 :
111  tmp = x;
112  x = 127 - y;
113  y = tmp;
114  break;
115  case BC_MODULE_LCD_ROTATION_180:
116  x = 127 - x;
117  y = 127 - y;
118  break;
119  case BC_MODULE_LCD_ROTATION_270:
120  tmp = y;
121  y = 127 - x;
122  x = tmp;
123  break;
124  case BC_MODULE_LCD_ROTATION_0:
125  break;
126  default:
127  break;
128  }
129 
130  // Skip mode byte + addr byte
131  uint32_t byteIndex = 2;
132  // Skip lines
133  byteIndex += y * 18;
134  // Select column byte
135  byteIndex += x / 8;
136 
137  uint8_t bitMask = 1 << (7 - (x % 8));
138 
139  if(!value)
140  {
141  self->framebuffer[byteIndex] |= bitMask;
142  }
143  else
144  {
145  self->framebuffer[byteIndex] &= ~bitMask;
146  }
147 }

149 int bc_module_lcd_draw_char(int left, int top, uint8_t ch)
150 {
151  bc_module_lcd_t *self = &_bc_module_lcd;
152 
153  const tFont *font = self->font;
154 
155  int w = 0;
156  uint8_t h = 0;
157  uint16_t i;
158  uint16_t x;
159  uint16_t y;
160  uint8_t bytes;
161 
162  for (i = 0; i < font->length; i++)
163  {
164  if (font->chars[i].code == ch)
165  {
166  w = font->chars[i].image->width;
167  h = font->chars[i].image->heigth;
168 
169  bytes = (w + 7) / 8;
170 
171  for (y = 0; y < h; y++)
172  {
173  for (x = 0; x < w; x++)
174  {
175  uint32_t byteIndex = x / 8;
176  byteIndex += y * bytes;
177 
178  uint8_t bitMask = 1 << (7 - (x % 8));
179 
180  if (font->chars[i].image->image[byteIndex] & bitMask)
181  {
182  bc_module_lcd_draw_pixel(left + x, top + y, false);
183  }
184  else
185  {
186  bc_module_lcd_draw_pixel(left + x, top + y, true);
187  }
188  }
189  }
190  }
191  }
192 
193  return w;
194 }
